# 1. Escolhe a imagem base do Node.js
FROM node:20-alpine

# 2. Cria o diretório de trabalho dentro do container
WORKDIR /usr/src/app

# 3. Copia os arquivos de dependências primeiro (otimiza o cache)
COPY package*.json ./

# 4. Instala as dependências da aplicação
RUN npm install

# 5. Copia o restante dos arquivos (server.js, html, etc.)
COPY . .

# 6. Expõe a porta que o seu server.js utiliza (ex: 3000)
EXPOSE 3000

# 7. Comando para iniciar a aplicação
CMD ["node", "server.js"]
